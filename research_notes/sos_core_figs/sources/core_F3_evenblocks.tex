% sos2cayley evenblocks.sos --rename a=a,!a=b
% Hand-placed on the shared grid (see figures.md): the same shape as the squares,
% grown to a 2x3 rectangle (the three swap pairs are its columns and its top row),
% with [a.a] between [a] and [b] -- which is what turns its b-edge into a
% neighbour hop instead of an edge across the picture. Styles are centralized in
% the \tikzset below: edit a style to restyle, edit a coordinate to move a node.
\documentclass[tikz,border=4pt]{standalone}
\usetikzlibrary{arrows.meta,shapes.misc}
\begin{document}
\begin{tikzpicture}[
  class/.style     = {draw, thin, rounded corners=2pt, inner sep=3pt,
                      minimum width=10mm, minimum height=7mm},
  root/.style      = {},          % the root needs no ink: the init stub says it
  init/.style      = {semithick, -{Stealth[length=4pt,width=3pt]}},
  tree/.style      = {semithick, -{Stealth[length=4pt,width=3pt]}},
  nontree/.style   = {thin, -{Stealth[length=4pt,width=3pt]}},
  lbl/.style       = {font=\small, inner sep=1.5pt, fill=white},
  pairs/.style     = {anchor=north, font=\small},
]

  \node[class,root] (eps) at (2.2,2.6)  {$\varepsilon$};
  \node[class]      (a)   at (0.0,0.9)  {$a$};
  \node[class] (aa)  at (2.2,0.9)  {$a{\cdot}a$};
  \node[class] (b)   at (4.4,0.9)  {$b$};
  \node[class]      (ab)  at (0.0,-1.7) {$a{\cdot}b$};
  \node[class]      (ba)  at (4.4,-1.7) {$b{\cdot}a$};
  \node[class] (aba) at (0.0,-4.3) {$a{\cdot}b{\cdot}a$};
  \node[class] (bab) at (4.4,-4.3) {$b{\cdot}a{\cdot}b$};

  % the root is the adjoined identity: a source, marked like an initial state
  \draw[init] (2.2,3.5) -- (eps);

  \draw[tree]    (eps) to              node[lbl] {$a$} (a);
  \draw[tree]    (eps) to              node[lbl] {$b$} (b);
  \draw[tree]    (a)   to[bend left=12] node[lbl] {$a$} (aa);   % swap pair 1
  \draw[nontree] (aa)  to[bend left=12] node[lbl] {$a$} (a);
  \draw[nontree] (aa)  to               node[lbl] {$b$} (b);
  \draw[tree]    (b)   to[bend left=12] node[lbl] {$a$} (ba);   % swap pair 2
  \draw[nontree] (ba)  to[bend left=12] node[lbl] {$a$} (b);
  \draw[nontree] (b)   to[loop right]   node[lbl] {$b$} (b);
  \draw[tree]    (a)   to               node[lbl] {$b$} (ab);
  \draw[tree]    (ab)  to[bend left=12] node[lbl] {$a$} (aba);  % swap pair 3
  \draw[nontree] (aba) to[bend left=12] node[lbl] {$a$} (ab);
  \draw[nontree] (ab)  to[loop left]    node[lbl] {$b$} (ab);
  \draw[tree]    (ba)  to               node[lbl] {$b$} (bab);
  \draw[nontree] (aba) to               node[lbl] {$b$} (bab);
  \draw[nontree] (bab) to[loop right]   node[lbl] {$a,b$} (bab);

  \node[pairs] at (2.2,-5.7)
    {$P = \{\, ([b],[b]),\; ([a{\cdot}b],[b]),\; ([b{\cdot}a],[a{\cdot}b{\cdot}a]),\;
      ([a{\cdot}b{\cdot}a],[a{\cdot}b{\cdot}a]),\; ([b{\cdot}a{\cdot}b],[b]),\;
      ([b{\cdot}a{\cdot}b],[a{\cdot}b{\cdot}a]) \,\}$};
\end{tikzpicture}
\end{document}
