against this very file, so a fresh drop runs only the languages it does not yet hold:

```

    --done ../reference/census/sweep_results.csv -o tests/sosl/logs/cluster/sweep.cmds
cluster/oarrun.sh --timeout 300 --cores 4 --split 250 sosl/tests/sosl/logs/cluster/sweep.cmds
cluster/reap_until.sh "$RUN"
python3 -m tests.sosl.merge_drops --key case_id,config_id -o <merged> \
    ../reference/census/sweep_results.csv ../logs/cluster/<RUN>/results.csv
```
